## CREATE DATABASE
* CREATE DATABASE
  - creates a new database
{% highlight SQL %}
CREATE DATABASE GameDB;
{% endhighlight %}

* CREATE TABLE
  - creates a new table
  - defines column and constraints
{% highlight SQL %}
CREATE TABLE accounts(
	accountId INTEGER NOT NULL,
	accountName VARCHAR(10) NOT NULL,
	coins INTEGER DEFAULT 0,
	createdTime DATETIME
);
{% endhighlight %}

* DROP TABLE
  - removes the table
{% highlight SQL %}
DROP TABLE accounts;
{% endhighlight %}

* ALTER TABLE
  - changes the table
  - ADD: add new column

{% highlight SQL %}

ADD lastEnterTime DATETIME;
{% endhighlight %}

* CONSTRAINT
  - defines the name of constraint
  - it makes easier to remove the contraint

{% highlight SQL %}
ALTER TABLE accounts
ADD CONSTRAINT PK_Account PRIMARY KEY(accountId);
{% endhighlight %}

{% highlight SQL %}
ALTER TABLE accounts
DROP CONSTRAINT PK_Account;
{% endhighlight %}

## INDEX
  - for regulation, you can use `INDEX` to index data

* Clustered
  - data is saved by order(ex. alpabet, number, etc.)
  - normally, it is primary key

* Non-Clustered
  - data is saved by index

* CREATE INDEX
  - makes new non-clustered index

{% highlight SQL %}
CREATE INDEX i1 ON accounts(accountNAme);
{% endhighlight %}
